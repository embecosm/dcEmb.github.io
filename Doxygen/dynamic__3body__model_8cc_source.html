73 Eigen::MatrixXd dynamic_3body_model::eval_generative(
74  const Eigen::VectorXd& parameters,
75  const parameter_location_3body& parameter_locations,
76  const int& timeseries_length, const int& num_bodies) {
77  Eigen::MatrixXd output =
78  Eigen::MatrixXd::Zero(timeseries_length, parameters.size());
79  double h = 0.001;
80  Eigen::VectorXd state = parameters;
81  output.row(0) = state;
82  std::function<Eigen::VectorXd(Eigen::VectorXd)> dfdt =
83  std::bind(&dynamic_3body_model::differential_eq, this,
84  std::placeholders::_1, num_bodies);
85  for (int i = 1; i < timeseries_length; i++) {
86  for (int j = 0; j < 10; j++) {
87  Eigen::VectorXd state_delta = utility::rungekutta(dfdt, state, h);
88  state = state + state_delta;
89  }
90  output.row(i) = state;
91  }
92  return output;
93 }

109 Eigen::VectorXd dynamic_3body_model::differential_eq(
110  const Eigen::VectorXd& state_in, const int& num_bodies) {
111  Eigen::VectorXd state_var = state_in;
112  double G = 1; // Gravitational Constant
113  // Convert our state vector to a matrix where each col is a planet
114  Eigen::Map<Eigen::MatrixXd> state(state_var.data(), 7, num_bodies);
115  // Initialise return matrix
116  Eigen::MatrixXd return_matrix =
117  Eigen::MatrixXd::Zero(state.rows(), state.cols());
118  for (int i = 0; i < state.cols(); i++) { // For each planet in turn
119  return_matrix(1, i) = state(4, i); // Update position x by velocity x
120  return_matrix(2, i) = state(5, i); // Update position y by velocity y
121  return_matrix(3, i) = state(6, i); // Update position z by velocity z
122  for (int j = 0; j < state.cols(); j++) { // For each other planet
123  if (i == j) {
124  continue; // Skip if comparing a planet to itself
125  }
126  double distancex = state(1, j) - state(1, i); // x distance
127  double distancey = state(2, j) - state(2, i); // y distance
128  double distancez = state(3, j) - state(3, i); // z distance
129  double distance_euclidian =
130  sqrt((distancex * distancex) + (distancey * distancey) +
131  (distancez * distancez)); // euclidian distance
132  // Update the x velocity by the x acceleration calculated at this point
133  return_matrix(4, i) +=
134  (G * state(0, j) * distancex) / pow(distance_euclidian, 3);
135  // Update the y velocity by the y acceleration calculated at this point
136  return_matrix(5, i) +=
137  (G * state(0, j) * distancey) / pow(distance_euclidian, 3);
138  // Update the z velocity by the z acceleration calculated at this point
139  return_matrix(6, i) +=
140  (G * state(0, j) * distancez) / pow(distance_euclidian, 3);
141  }
142  }
143  // Convert return matrix to vector
144  Eigen::Map<Eigen::VectorXd> return_state(
145  return_matrix.data(), return_matrix.rows() * return_matrix.cols());
146  return return_state;
147 }
