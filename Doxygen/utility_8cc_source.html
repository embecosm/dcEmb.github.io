49 double utility::logdet(const Eigen::MatrixXd& mat) {
50  double tol = 1e-16;
51  if (mat.cols() != mat.rows()) {
52  std::cout << "Error: Expected Square Matrix in logdet";
53  throw;
54  }
55  if (mat.isDiagonal()) {
56  Eigen::VectorXd eig_values = mat.diagonal();
57  Eigen::VectorXd pos_eig_values = eig_values.unaryExpr([tol](double x) {
58  return ((x > tol) && (x < 1 / tol)) ? log(x) : 0.0;
59  });
60  return pos_eig_values.sum();
61  } else {
62  Eigen::BDCSVD<Eigen::MatrixXd> svd;
63  svd.setThreshold(1e-6);
64  svd.compute(mat, Eigen::ComputeThinV | Eigen::ComputeThinU);
65  Eigen::VectorXd singular_values = svd.singularValues();
66  Eigen::VectorXd log_singular_values =
67  singular_values.unaryExpr([tol](double x) {
68  return ((x > tol) && (x < 1 / tol)) ? log(x) : 0.0;
69  });
70  return log_singular_values.sum();
71  }
72 }

79  std::function<Eigen::VectorXd(Eigen::VectorXd)> func, Eigen::VectorXd& vars,
80  Eigen::MatrixXd& transform) {
81  Eigen::VectorXd base = func(vars);
82  Eigen::MatrixXd out_matrix = Eigen::MatrixXd::Zero(base.size(), vars.size());
83  double dx = exp(-8.0);
84 #pragma omp parallel
85  {
86 #pragma omp for schedule(static)
87  for (int i = 0; i < vars.size(); i++) {
88  Eigen::VectorXd vars_tmp = vars + (transform.col(i) * dx);
89  Eigen::VectorXd modified = func(vars_tmp);
90  out_matrix.col(i) = (modified - base) / dx;
91  }
92  }
93  return out_matrix;

113 SparseMD utility::permute_kron_matrix(const SparseMD& matrix,
114  const Eigen::VectorXi& new_order,
115  const Eigen::VectorXi& cur_order_size) {
116  // Permute the order of elements in a kronecker matrix. Each value on each
117  // axis in the kronecker matrix represents a position in new_order.size()
118  // dimensions. Calculate this position for the canonical ordering
119  // ([1, ..., new_order.size() -1]), and then recalculate it for the new
120  // ordering (new_order), and apply this to both matrix dimensions.
121  Eigen::VectorXi pos_vector = Eigen::VectorXi::Ones(cur_order_size.size());
122  Eigen::VectorXi pos_vector_new = Eigen::VectorXi::Ones(cur_order_size.size());
123  int size_iterator = 1;
124  int size_iterator_new = 1;
125  Eigen::VectorXi new_order_size = cur_order_size(new_order);
126  // Calculate the values of the vectors converting position to co-ordinates
127  for (int i = 1; i < cur_order_size.size(); i++) {
128  pos_vector(i) = cur_order_size(i - 1) * size_iterator;
129  pos_vector_new(i) = new_order_size(i - 1) * size_iterator_new;
130  size_iterator = pos_vector(i);
131  size_iterator_new = pos_vector_new(i);
132  }
133 
134  // Calculate the new positions with the changed dimensions
135  Eigen::VectorXi perm_vector = Eigen::VectorXi(cur_order_size.prod());
136  for (int i = 0; i < cur_order_size.prod(); i++) {
137  Eigen::VectorXi nums = Eigen::VectorXi::Zero(cur_order_size.size());
138  int size_iterator2 = i;
139  for (int j = cur_order_size.size() - 1; j > 0; j--) {
140  nums(j) = size_iterator2 / pos_vector(j);
141  size_iterator2 = size_iterator2 % pos_vector(j);
142  }
143  nums(0) = size_iterator2;
144 
145  perm_vector(i) = nums(new_order).transpose() * pos_vector_new;
146  }
147 
148  // Reorder the elements in the matrix
149  // NB: Test carefully any changes to this.
150  SparseMD matrix2(matrix.rows(), matrix.cols());
151  // Since we are only permuting transition matricies, we have a pretty good
152  // idea of how populated they are, reserve in advance to improve performance
153  matrix2.reserve(Eigen::VectorXi::Constant(matrix.cols(), 3));
154 
155  for (int i = 0; i < matrix.outerSize(); i++) {
156  for (SparseMD::InnerIterator it(matrix, i); it; ++it) {
157  matrix2.insert(perm_vector(it.row()), perm_vector(it.col())) = it.value();
158  }
159  }
160 
161  return matrix2;
162 }

250  int num_rows = mat.rows();
251  int num_cols = mat.cols();
252  Eigen::BDCSVD<Eigen::MatrixXd> svd;
253  svd.compute(mat, Eigen::ComputeThinV | Eigen::ComputeThinU);
254  int rank = svd.rank();
255 
256  Eigen::MatrixXd u = Eigen::MatrixXd::Zero(rank, rank);
257  u.col(0) = mat.col(0);
258  int j_it = 0;
259  for (int i = 1; i < rank; i++) {
260  Eigen::MatrixXd pinv = u.completeOrthogonalDecomposition().pseudoInverse();
261  Eigen::VectorXd u_n = mat.col(i);
262  u_n = u_n - (u * pinv * u_n);
263  if (!(u_n.norm() > exp(-32))) {
264  throw std::runtime_error(
265  "Currently unspecified behaviour in utility::orth - norm out of "
266  "range");
267  return Eigen::MatrixXd::Zero(1, 1);
268  }
269  u.col(i) = u_n;
270  }
271  return u;
272 }

345  (reduced_parameter_expectations.transpose() *

366  return_vector[0] = free_energy;
367  return_vector[1] = conditional_reduced_expectations;
368  return_vector[2] = conditional_reduced_covariances;
369  return return_vector;

382  const double& tol) {

384  diag = diag * tol;

388  ret_mat.diagonal() =
