82 template <typename Derived, typename T = typename Derived::Scalar>
83 Derived softmax(const Eigen::MatrixBase<Derived>& mat) {
84  T max = mat.maxCoeff();
85  Derived mat_norm = mat.unaryExpr([max](T x) { return (x - max); });
86  T exp_mat_sum = mat_norm.array().exp().sum();
87  if (exp_mat_sum == INFINITY) {
88  throw std::runtime_error("vector values too large");
89  return Derived::Zero(1);
90  }
91  return mat_norm.unaryExpr([exp_mat_sum](T x) { return (exp(x) / exp_mat_sum); });
92 }

158  std::random_device rd{};

173 template <typename Derived, typename T = typename Derived::Scalar>
174 int selrnd(const Eigen::MatrixBase<Derived>& dist) {
175  if ((dist.rows() != 1) && (dist.cols() != 1)) {
176  throw std::runtime_error("error: dist must be a vector of numbers");
177  return -1;
178  }
179 
180  std::random_device rd{};
181  std::mt19937 gen{rd()};
182  std::uniform_real_distribution<T> d{0, 1};
183  T rnd = d(gen);
184 
185  T cum_sum = 0;
186

188  cum_sum += dist(i - 1);

192  }

201 template <typename Derived, typename T = typename Derived::Scalar>
202 Derived gradient(const Derived& vec) {
203  if ((vec.rows() != 1) && (vec.cols() != 1)) {
204  throw std::runtime_error("error: gradient must take a vector of numbers");
205  }
206  int sz = vec.size();
207  Derived result_vector(sz);
208  for (Eigen::Index i = 1; i < sz - 1; i++) {
209  result_vector(i) = 0.5 * ((T)vec(i + 1) - (T)vec(i - 1));
210  }
211  result_vector(0) = vec(1) - vec(0);
212  result_vector(sz - 1) = vec(sz - 1) - vec(sz - 2);
213  return result_vector;
214 }
