93  Eigen::VectorXd lat = Eigen::VectorXd(num_countries);
94  Eigen::VectorXd lon = Eigen::VectorXd(num_countries);
95  Eigen::VectorXd pop = Eigen::VectorXd(num_countries);
96  Eigen::VectorXd ones = Eigen::VectorXd::Ones(num_countries);
97  for (int i = 0; i < num_countries; i++) {
98  lat(i) = (countries.at(i).latitude * 2 * M_PI / 360);
99  lon(i) = (countries.at(i).longitude * 2 * M_PI / 360);
100  pop(i) = log(countries.at(i).pop);
101  }
102  Eigen::MatrixXd design_matrix_tmp = Eigen::MatrixXd(5, 10);
103  design_matrix_tmp.col(0) = ones;
104  design_matrix_tmp.col(1) = pop;
105  for (int i = 1; i < 5; i++) {
106  design_matrix_tmp.col(2 * i) =
107  lon.unaryExpr([i](double x) { return sin(i * x); });
108  design_matrix_tmp.col(2 * i + 1) =
109  lat.unaryExpr([i](double x) { return sin(i * x); });
110  }
111  design_matrix_tmp = utility::orth(design_matrix_tmp);
112  Eigen::MatrixXd design_matrix = Eigen::MatrixXd(5, 5);
113  for (int i = 0; i < 5; i++) {
114  design_matrix.col(i) =
115  design_matrix_tmp.col(i) /

159 Eigen::VectorXd default_prior_expectations() {
160  Eigen::VectorXd default_prior_expectation = Eigen::VectorXd::Zero(28);
161  default_prior_expectation << log(4), // number of initial cases
162  log(8), // size of population with mixing
163  -log(4), // initial proportion
164  -log(3), // P(going home | work)
165  -log(32), // social distancing threshold
166  log(16) - log(100000), // bed availability threshold (per capita)
167  log(4), // effective number of contacts: home
168  log(48), // effective number of contacts: work
169  -log(3), // P(transmission | infectious)
170  log(4), // infected (pre-contagious) period
171  log(4), // contagious period
172  log(16), // time until symptoms
173  -log(32), // P(severe symptoms | symptomatic)
174  log(8), // symptomatic period
175  log(10), // period in CCU
176  -log(2), // P(fatality | CCU)
177  -log(8), // P(fatality | home)
178  -log(10000), // test, track and trace
179  log(2), // testing latency (months)
180  log(4), // test delay (days)
181  log(8), // test selectivity (for infection)
182  log(4) - log(10000), // sustained testing
183  log(4) - log(10000), // baseline testing
184  log(16), // period of immunity
185  log(2), // period of exemption
186  -log(2), // proportion of people not susceptible
187  -log(2), // proportion with innate immunity
188  log(16); // testing buildup
189  return default_prior_expectation;
190 }

214  precise, // symptomatic period

216  precise, // P(fatality | CCU)
217  precise, // P(fatality | home)
