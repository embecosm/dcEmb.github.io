33 void dynamic_model::invert_model() {
34  Eigen::MatrixXd response_vars_fs = this->get_observed_outcomes();
35  int num_response_total = this->num_response_vars * this->num_samples;
36  if (!num_response_vars) {
37  num_response_vars = select_response_vars.size();
38  }
39  Eigen::MatrixXd initial_state = Eigen::MatrixXd::Zero(1, 1);
40  int dt = 1;
41  std::vector<SparseMD> precision_comp(num_response_vars);
42  int num_precision_comp = num_response_vars;
43  for (int i = 0; i < num_response_vars; i++) {
44  SparseMD prec(num_response_total, num_response_total);
45  for (int j = i * num_samples; j < i * num_samples + num_samples; j++) {
46  prec.coeffRef(j, j) = 1;
47  }
48  precision_comp[i] = prec;
49  }
50  Eigen::VectorXd prior_p_e = this->prior_parameter_expectations;
51  Eigen::MatrixXd prior_p_c = this->prior_parameter_covariances;
52  Eigen::VectorXd prior_h_e = this->prior_hyper_expectations;
53  Eigen::MatrixXd inv_prior_h_c = this->prior_hyper_covariances;
54  inv_prior_h_c = utility::inverse_tol(inv_prior_h_c);
55  // Note: different SVD implementations may produces different but
56  // equally valid decomps of U and V.
57  Eigen::BDCSVD<Eigen::MatrixXd> svd;
58  svd.setThreshold(std::numeric_limits<double>::epsilon() * 64);
59  svd.compute(prior_p_c, Eigen::ComputeThinV | Eigen::ComputeThinU);
60  Eigen::MatrixXd singular_vec = svd.matrixU();
61 
62  int num_parameters_eff = svd.nonzeroSingularValues();
63  // Reduce parameter space
64  prior_p_c = singular_vec.transpose() * prior_p_c * singular_vec;
65  Eigen::MatrixXd inv_prior_p_c = utility::inverse_tol(prior_p_c);
66 
67  Eigen::VectorXd likel_p_e = Eigen::VectorXd::Zero(num_parameters_eff);
68  Eigen::VectorXd conditional_p_e = prior_p_e;
69  double ascent_rate = -4.0;
70 
71  Eigen::VectorXd h_estimate = prior_h_e;
72  Eigen::VectorXd p_estimate = Eigen::VectorXd::Zero(prior_p_e.size());
73 
74  Eigen::VectorXd current_p_estimate;
75  Eigen::MatrixXd current_p_con_cov_estimate;
76  Eigen::VectorXd current_h_estimate;
77  double current_free_energy = -INFINITY;
78  double initial_free_energy = -INFINITY;
79  int criterion = 4;
80  int num_success = 0;
81  std::function<Eigen::VectorXd(Eigen::VectorXd)> forward_model_function =
82  this->get_forward_model_function();
83 
84  Eigen::VectorXd dFdp;
85  Eigen::MatrixXd dFdpp;
86  for (int i = 0; i < this->max_invert_it; i++) {
87  auto start = std::chrono::high_resolution_clock::now();
88 
89  // Runtime polymorphism - we have to use this->forward model to make
90  // sure we've calling the implementation of forward model method of
91  // inheriting classes
92  Eigen::MatrixXd predicted_f = forward_model_function(conditional_p_e);
93  Eigen::MatrixXd dfdp =
94  utility::diff(forward_model_function, conditional_p_e, singular_vec);
95  // Check for stability. Is the Infinity Norm in resonable bounds?
96  double norm = dfdp.cwiseAbs().rowwise().sum().maxCoeff();
97  bool revert = (norm > 1e32);
98  if (revert) {
99  throw std::runtime_error("revert hit");
100  }
101  Eigen::VectorXd prediction_error = response_vars_fs - predicted_f;
102  Eigen::VectorXd hyper_error;
103  // J == -dfdp with no confounds present
104  Eigen::MatrixXd J = -dfdp;
105  Eigen::MatrixXd i_cov_comp;
106  Eigen::MatrixXd conditional_p_cov;
107  Eigen::MatrixXd conditional_h_cov;
108  for (int k = 0; k < 8; k++) {
109  i_cov_comp =
110  Eigen::MatrixXd::Zero(num_response_total, num_response_total);
111  for (int l = 0; l < num_precision_comp; l++) {
112 
113  i_cov_comp =
114  i_cov_comp + (precision_comp[l] * (exp(-32) + exp(h_estimate(l))));
115  }
116  Eigen::MatrixXd cov_comp = utility::inverse_tol(i_cov_comp);
117  Eigen::MatrixXd i_conditional_p_cov =
118  (J.transpose() * i_cov_comp * J) + inv_prior_p_c;
119  conditional_p_cov = utility::inverse_tol(i_conditional_p_cov);
120 
121  std::vector<SparseMD> P_mstep_op(num_response_vars);
122  std::vector<SparseMD> PS_mstep_op(num_response_vars);
123  std::vector<SparseMD> JPJ_mstep_op(num_response_vars);
124 
125  SparseMD cov_comp_sparse = cov_comp.sparseView();
126  SparseMD J_sparse = J.sparseView();
127  for (int l = 0; l < num_response_vars; l++) {
128  SparseMD P = (precision_comp[l] * exp(h_estimate(l))).eval();
129  SparseMD PS = P * cov_comp_sparse;
130  SparseMD JPJ = J_sparse.transpose() * P * J_sparse;
131  P_mstep_op[l] = P;
132  PS_mstep_op[l] = PS;
133  JPJ_mstep_op[l] = JPJ;
134  }
135 
136  Eigen::VectorXd dFdh = Eigen::VectorXd::Zero(num_response_vars);
137  Eigen::MatrixXd dFdhh =
138  Eigen::MatrixXd::Zero(num_response_vars, num_response_vars);
139  for (int l = 0; l < num_response_vars; l++) {
140  Eigen::MatrixXd ePe =
141  (prediction_error.transpose() * P_mstep_op[l] * prediction_error) /
142  2;
143  dFdh(l) = (PS_mstep_op[l]).toDense().trace() / 2 - ePe.value() -
144  (conditional_p_cov * JPJ_mstep_op[l]).trace() / 2;
145  for (int m = 0; m < num_response_vars; m++) {
146  dFdhh(l, m) =
147  -utility::SparseProductTrace(PS_mstep_op[l], PS_mstep_op[m]) / 2;
148  dFdhh(m, l) = dFdhh(l, m);
149  }
150  }
151  hyper_error = (h_estimate - prior_h_e);
152  dFdh = dFdh - (inv_prior_h_c * hyper_error);
153  dFdhh = dFdhh - inv_prior_h_c;
154  conditional_h_cov = utility::inverse_tol(-dFdhh);
155  Eigen::VectorXd dh = utility::dx(dFdhh, dFdh, ascent_rate);
156  dh = (dh.unaryExpr(
157  [](double x) { return std::min(std::max(x, -1.0), 1.0); })).eval();
158  h_estimate = h_estimate + dh;
159  double dFh = dFdh.transpose() * dh;
160 
161  if (dFh < 1e-2) {
162  break;
163  }
164  }
165 
166  double L1 = (utility::logdet(i_cov_comp) -
167  prediction_error.transpose() * i_cov_comp * prediction_error -
168  num_response_total * log(8 * atan(1.0))) /
169  2;
170  double L2 = (utility::logdet(inv_prior_p_c * conditional_p_cov) -
171  p_estimate.transpose() * inv_prior_p_c * p_estimate) /
172  2;
173  double L3 = (utility::logdet(inv_prior_h_c * conditional_h_cov) -
174  hyper_error.transpose() * inv_prior_h_c * hyper_error) /
175  2;
176  double free_energy_var = L1 + L2 + L3;
177 
178  if (initial_free_energy == -INFINITY) {
179  initial_free_energy = free_energy_var;
180  }
181  auto end = std::chrono::high_resolution_clock::now();
182  auto duration =
183  std::chrono::duration_cast<std::chrono::microseconds>(end - start);
184  if (i > 0) {
185  std::cout << "actual: " << free_energy_var - current_free_energy
186  << " microsec: " << (duration.count()) << '\n';
187  }
188 
189  std::string str;
190  if ((free_energy_var > current_free_energy) | (i < 2)) {
191  current_p_estimate = p_estimate;
192  current_p_con_cov_estimate = conditional_p_cov;
193  current_h_estimate = h_estimate;
194  current_free_energy = free_energy_var;
195 
196  dFdp = (-(J.transpose() * i_cov_comp * prediction_error) -
197  inv_prior_p_c * p_estimate);
198  dFdpp = (-(J.transpose() * i_cov_comp * J) - inv_prior_p_c);
199  ascent_rate = std::min(ascent_rate + 0.5, 4.0);
200  str = "EM:(+) " + std::to_string(i + 1) + " ";
201  } else {
202  p_estimate = current_p_estimate;
203  conditional_p_cov = current_p_con_cov_estimate;
204  h_estimate = current_h_estimate;
205  ascent_rate = std::min(ascent_rate - 2, -4.0);
206  str = "EM:(-) " + std::to_string(i + 1) + " ";
207  }
208  Eigen::VectorXd dp = utility::dx(dFdpp, dFdp, ascent_rate);
209  p_estimate = p_estimate + dp;
210 
211  conditional_p_e =
212  prior_p_e +
213  singular_vec * p_estimate(Eigen::seq(0, num_parameters_eff - 1));
214  double dF = dFdp.transpose() * dp;
215 
216  std::cout << str << "F: " << current_free_energy - initial_free_energy
217  << ' ' << "dF predicted: " << dF << ' ';
218  if (dF < 1e-1) {
219  num_success += 1;
220  if (num_success == criterion) {
221  auto end = std::chrono::high_resolution_clock::now();
222  auto duration =
223  std::chrono::duration_cast<std::chrono::microseconds>(end - start);
224  std::cout << "actual: " << free_energy_var - current_free_energy
225  << " microsec: " << (duration.count()) << '\n';
226  std::cout << "convergence" << '\n';
227  break;
228  }
229  } else {
230  num_success = 0;
231  }
232  }
233  this->conditional_parameter_expectations =
234  prior_p_e +
235  singular_vec * p_estimate(Eigen::seq(0, num_parameters_eff - 1));
236  this->conditional_parameter_covariances =
237  singular_vec * current_p_con_cov_estimate * singular_vec.transpose();
238  this->conditional_hyper_expectations = current_h_estimate;
239  this->free_energy = current_free_energy;
240  return;
241 }
